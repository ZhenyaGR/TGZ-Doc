# –ü–∞–≥–∏–Ω–∞—Ü–∏—è (Pagination)

–ö–ª–∞—Å—Å `Pagination` ‚Äî —ç—Ç–æ –º–æ—â–Ω—ã–π –∏–Ω—Å—Ç—Ä—É–º–µ–Ω—Ç –¥–ª—è –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–≥–æ —Ä–∞–∑–±–∏–µ–Ω–∏—è –±–æ–ª—å—à–∏—Ö —Å–ø–∏—Å–∫–æ–≤ –∫–Ω–æ–ø–æ–∫ –Ω–∞ —Å—Ç—Ä–∞–Ω–∏—Ü—ã. –í–º–µ—Å—Ç–æ —Ç–æ–≥–æ —á—Ç–æ–±—ã –≤—Ä—É—á–Ω—É—é –≤—ã—Å—á–∏—Ç—ã–≤–∞—Ç—å —Å–º–µ—â–µ–Ω–∏—è (offset) –∏ —Ä–∏—Å–æ–≤–∞—Ç—å –∫–Ω–æ–ø–∫–∏ –Ω–∞–≤–∏–≥–∞—Ü–∏–∏ (`<`, `>`), –≤—ã –ø—Ä–æ—Å—Ç–æ –ø–µ—Ä–µ–¥–∞–µ—Ç–µ –º–∞—Å—Å–∏–≤ –≤—Å–µ—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤, –∞ –±–∏–±–ª–∏–æ—Ç–µ–∫–∞ –¥–µ–ª–∞–µ—Ç –æ—Å—Ç–∞–ª—å–Ω–æ–µ.

## –ü—Ä–∏–Ω—Ü–∏–ø —Ä–∞–±–æ—Ç—ã

1.  –í—ã —Å–æ–∑–¥–∞–µ—Ç–µ –º–∞—Å—Å–∏–≤ –≤—Å–µ—Ö –∫–Ω–æ–ø–æ–∫ (–Ω–∞–ø—Ä–∏–º–µ—Ä, 50 —Ç–æ–≤–∞—Ä–æ–≤).
2.  –ü–µ—Ä–µ–¥–∞–µ—Ç–µ –∏—Ö –≤ `Pagination`.
3.  –ù–∞—Å—Ç—Ä–∞–∏–≤–∞–µ—Ç–µ –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏–µ (—Å–∫–æ–ª—å–∫–æ –∫–Ω–æ–ø–æ–∫ –Ω–∞ —Å—Ç—Ä–∞–Ω–∏—Ü–µ, —Å—Ç–∏–ª—å –Ω–∞–≤–∏–≥–∞—Ü–∏–∏).
4.  –£–∫–∞–∑—ã–≤–∞–µ—Ç–µ —Ç–µ–∫—É—â—É—é —Å—Ç—Ä–∞–Ω–∏—Ü—É.
5.  –ú–µ—Ç–æ–¥ `create()` –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –≥–æ—Ç–æ–≤—ã–π –º–∞—Å—Å–∏–≤ –∫–Ω–æ–ø–æ–∫ –¥–ª—è –º–µ—Ç–æ–¥–∞ `inlineKbd()`.

## –ë–∞–∑–æ–≤–æ–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ

–°–æ–∑–¥–∞–¥–∏–º –ø—Ä–æ—Å—Ç–æ–π —Å–ø–∏—Å–æ–∫ –∏–∑ 20 –∫–Ω–æ–ø–æ–∫ –∏ –æ—Ç–æ–±—Ä–∞–∑–∏–º –ø–µ—Ä–≤—É—é —Å—Ç—Ä–∞–Ω–∏—Ü—É.

```php

use ZenithGram\ZenithGram\Button; // –ò—Å–ø–æ–ª—å–∑—É–µ–º —Å–æ–≤—Ä–µ–º–µ–Ω–Ω—ã–π –∫–ª–∞—Å—Å –∫–Ω–æ–ø–æ–∫

// 1. –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —Ç–µ—Å—Ç–æ–≤—ã–π –º–∞—Å—Å–∏–≤ –∫–Ω–æ–ø–æ–∫

    // –°–æ–∑–¥–∞–µ–º callback-–∫–Ω–æ–ø–∫—É –¥–ª—è –∫–∞–∂–¥–æ–≥–æ —Ç–æ–≤–∞—Ä–∞

    ->setPerPage(5)             // 5 —Ç–æ–≤–∞—Ä–æ–≤ –Ω–∞ —Å—Ç—Ä–∞–Ω–∏—Ü–µ

    ->send();
```

## –û–±—Ä–∞–±–æ—Ç–∫–∞ –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏–π (–ò–Ω—Ç–µ—Ä–∞–∫—Ç–∏–≤–Ω–æ—Å—Ç—å)

–°–∞–º–æ–µ –≤–∞–∂–Ω–æ–µ –≤ –ø–∞–≥–∏–Ω–∞—Ü–∏–∏ ‚Äî –∑–∞—Å—Ç–∞–≤–∏—Ç—å –∫–Ω–æ–ø–∫–∏ –Ω–∞–≤–∏–≥–∞—Ü–∏–∏ —Ä–∞–±–æ—Ç–∞—Ç—å.
–ö–Ω–æ–ø–∫–∏ –Ω–∞–≤–∏–≥–∞—Ü–∏–∏ –≥–µ–Ω–µ—Ä–∏—Ä—É—é—Ç `callback_data` –≤ —Ñ–æ—Ä–º–∞—Ç–µ: `{prefix}{–Ω–æ–º–µ—Ä_—Å—Ç—Ä–∞–Ω–∏—Ü—ã}`.

–î–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ —É–¥–æ–±–Ω–æ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å –ø–ª–µ–π—Å—Ö–æ–ª–¥–µ—Ä `%n` (—á–∏—Å–ª–æ) –≤ –º–µ—Ç–æ–¥–µ `onCallback`.

```php
// –û–±—Ä–∞–±–æ—Ç—á–∏–∫ –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏—è —Å—Ç—Ä–∞–Ω–∏—Ü
// –õ–æ–≤–∏–º –ø–∞—Ç—Ç–µ—Ä–Ω "page_" + —á–∏—Å–ª–æ
$bot->onCallback('pagination_handler', 'page_%n')
    ->func(function(ZG $tg, int $page) use ($items) {
        $query_id = $tg->getQueryId();
        $tg->answerCallbackQuery($query_id); // –£–±–∏—Ä–∞–µ–º "—á–∞—Å–∏–∫–∏"

        // –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º –∫–ª–∞–≤–∏–∞—Ç—É—Ä—É –¥–ª—è –Ω–æ–≤–æ–π —Å—Ç—Ä–∞–Ω–∏—Ü—ã
        // –í–∞–∂–Ω–æ: –∑–¥–µ—Å—å –Ω—É–∂–Ω–æ –∑–∞–Ω–æ–≤–æ —Å–æ–∑–¥–∞—Ç—å –æ–±—ä–µ–∫—Ç –ø–∞–≥–∏–Ω–∞—Ü–∏–∏ —Å —Ç–µ–º–∏ –∂–µ –Ω–∞—Å—Ç—Ä–æ–π–∫–∞–º–∏
        $keyboard = $tg->pagination()
            ->setItems($items)
            ->setPerPage(5)
            ->setPage($page)     // –£—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –ø—Ä–∏—à–µ–¥—à—É—é —Å—Ç—Ä–∞–Ω–∏—Ü—É
            ->setPrefix('page_')
            ->create();

        // –†–µ–¥–∞–∫—Ç–∏—Ä—É–µ–º —Å–æ–æ–±—â–µ–Ω–∏–µ
        $tg->msg("–°–ø–∏—Å–æ–∫ —Ç–æ–≤–∞—Ä–æ–≤ (–°—Ç—Ä–∞–Ω–∏—Ü–∞ $page):")
           ->inlineKbd($keyboard)
           ->editText();
    });
```

## –ü—Ä–æ–¥–≤–∏–Ω—É—Ç–∞—è –Ω–∞—Å—Ç—Ä–æ–π–∫–∞

–ë–∏–±–ª–∏–æ—Ç–µ–∫–∞ –ø–æ–∑–≤–æ–ª—è–µ—Ç –≥–∏–±–∫–æ –Ω–∞—Å—Ç—Ä–∞–∏–≤–∞—Ç—å –≤–Ω–µ—à–Ω–∏–π –≤–∏–¥ –ø–∞–≥–∏–Ω–∞—Ç–æ—Ä–∞.

### 1. –†–µ–∂–∏–º –Ω–∞–≤–∏–≥–∞—Ü–∏–∏ (–°—Ç—Ä–µ–ª–∫–∏ vs –¶–∏—Ñ—Ä—ã)

–í—ã –º–æ–∂–µ—Ç–µ –≤—ã–±—Ä–∞—Ç—å, –∫–∞–∫ –±—É–¥—É—Ç –≤—ã–≥–ª—è–¥–µ—Ç—å –∫–Ω–æ–ø–∫–∏ –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏—è —Å—Ç—Ä–∞–Ω–∏—Ü: –∫–ª–∞—Å—Å–∏—á–µ—Å–∫–∏–µ —Å—Ç—Ä–µ–ª–∫–∏ –∏–ª–∏ —Ä—è–¥ –Ω–æ–º–µ—Ä–æ–≤.

```php
use ZenithGram\ZenithGram\Enums\PaginationMode;

// –†–µ–∂–∏–º: –°—Ç—Ä–µ–ª–∫–∏ (–ü–æ —É–º–æ–ª—á–∞–Ω–∏—é)
// –í–∏–¥: [ < ] [ > ]
$tg->pagination()->setMode(PaginationMode::ARROWS);

// –†–µ–∂–∏–º: –ù–æ–º–µ—Ä–∞ —Å—Ç—Ä–∞–Ω–∏—Ü
// –í–∏–¥: [ 1 ] [ 2 ] [ 3 ] [ 4 ] [ 5 ]
$tg->pagination()->setMode(PaginationMode::NUMBERS);
```

### 2. –°—Ç–∏–ª–∏–∑–∞—Ü–∏—è –Ω–æ–º–µ—Ä–æ–≤ —Å—Ç—Ä–∞–Ω–∏—Ü

–ï—Å–ª–∏ –≤—ã–±—Ä–∞–Ω —Ä–µ–∂–∏–º `NUMBERS`, –≤—ã –º–æ–∂–µ—Ç–µ –Ω–∞—Å—Ç—Ä–æ–∏—Ç—å —Å—Ç–∏–ª—å —Ü–∏—Ñ—Ä.

```php

// –≠–º–æ–¥–∑–∏: 1Ô∏è‚É£, 2Ô∏è‚É£, 3Ô∏è‚É£...

// –ö–∞—Å—Ç–æ–º–Ω—ã–π —Å—Ç–∏–ª—å (—á–µ—Ä–µ–∑ —Ñ—É–Ω–∫—Ü–∏—é)
$tg->pagination()->setNumberStyle(fn($page) => "[ –°—Ç—Ä. $page ]");
```

### 3. –†–∞—Å–ø–æ–ª–æ–∂–µ–Ω–∏–µ –∫–Ω–æ–ø–æ–∫ –Ω–∞–≤–∏–≥–∞—Ü–∏–∏ (Layout)

–£–ø—Ä–∞–≤–ª—è–π—Ç–µ —Ç–µ–º, –∫–∞–∫ –≥—Ä—É–ø–ø–∏—Ä—É—é—Ç—Å—è –∫–Ω–æ–ø–∫–∏ –Ω–∞–≤–∏–≥–∞—Ü–∏–∏, –µ—Å–ª–∏ –∏—Ö –º–Ω–æ–≥–æ (–Ω–∞–ø—Ä–∏–º–µ—Ä, "–ù–∞–∑–∞–¥", "–í–ø–µ—Ä–µ–¥", "–í –Ω–∞—á–∞–ª–æ", "–í –∫–æ–Ω–µ—Ü").

```php
use ZenithGram\ZenithGram\Enums\PaginationLayout;

// –í –æ–¥–Ω—É —Å—Ç—Ä–æ–∫—É (–ü–æ —É–º–æ–ª—á–∞–Ω–∏—é)
// [<<] [<] [>] [>>]
$tg->pagination()->setNavigationLayout(PaginationLayout::ROW);

// –†–∞–∑–¥–µ–ª–µ–Ω–∏–µ –Ω–∞ –¥–≤–µ —Å—Ç—Ä–æ–∫–∏
// –°—Ç—Ä–æ–∫–∞ 1: [<] [>]
// –°—Ç—Ä–æ–∫–∞ 2: [<<] [>>]
$tg->pagination()->setNavigationLayout(PaginationLayout::SPLIT);

// –£–º–Ω–∞—è –≥—Ä—É–ø–ø–∏—Ä–æ–≤–∫–∞ (SMART)
// –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ —Ä–∞–∑–±–∏–≤–∞–µ—Ç –Ω–∞ —Å—Ç—Ä–æ–∫–∏, –µ—Å–ª–∏ –∫–Ω–æ–ø–æ–∫ –±–æ–ª—å—à–µ 2-—Ö, —á—Ç–æ–±—ã –æ–Ω–∏ –Ω–µ –±—ã–ª–∏ —É–∑–∫–∏–º–∏.
$tg->pagination()->setNavigationLayout(PaginationLayout::SMART);
```

### 4. –ë—ã—Å—Ç—Ä–∞—è –Ω–∞–≤–∏–≥–∞—Ü–∏—è –∏ –≤—ã—Ö–æ–¥

```php
    // –í–∫–ª—é—á–∞–µ–º –∫–Ω–æ–ø–∫–∏ –±—ã—Å—Ç—Ä–æ–≥–æ –ø–µ—Ä–µ—Ö–æ–¥–∞

    ->setSigns('‚¨ÖÔ∏è –ù–∞–∑–∞–¥', '–í–ø–µ—Ä–µ–¥ ‚û°Ô∏è')
    
    // –î–æ–±–∞–≤–ª—è–µ–º –æ—Ç–¥–µ–ª—å–Ω—É—é –∫–Ω–æ–ø–∫—É "–í–µ—Ä–Ω—É—Ç—å—Å—è" –ø–æ–¥ –ø–∞–≥–∏–Ω–∞—Ü–∏–µ–π
    ->addReturnBtn('üîô –í—ã–π—Ç–∏ –≤ –º–µ–Ω—é', 'main_menu');
```

### 5. –§–æ—Ä–º–∞—Ç –∞–∫—Ç–∏–≤–Ω–æ–π —Å—Ç—Ä–∞–Ω–∏—Ü—ã

–í—ã –º–æ–∂–µ—Ç–µ –≤—ã–¥–µ–ª–∏—Ç—å —Ç–µ–∫—É—â—É—é —Å—Ç—Ä–∞–Ω–∏—Ü—É, —á—Ç–æ–±—ã –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å –ø–æ–Ω–∏–º–∞–ª, –≥–¥–µ –Ω–∞—Ö–æ–¥–∏—Ç—Å—è (–æ—Å–æ–±–µ–Ω–Ω–æ –∞–∫—Ç—É–∞–ª—å–Ω–æ –¥–ª—è —Ä–µ–∂–∏–º–∞ –Ω–æ–º–µ—Ä–æ–≤).

## –ü–æ–ª–Ω—ã–π –ø—Ä–∏–º–µ—Ä (–ú–∞–≥–∞–∑–∏–Ω "Pro")

–ü—Ä–∏–º–µ—Ä —Ä–µ–∞–ª–∏–∑–∞—Ü–∏–∏ —Å–ª–æ–∂–Ω–æ–≥–æ –∫–∞—Ç–∞–ª–æ–≥–∞ —Å –Ω—É–º–µ—Ä–∞—Ü–∏–µ–π —Å—Ç—Ä–∞–Ω–∏—Ü —Å–º–∞–π–ª–∏–∫–∞–º–∏, —É–º–Ω–æ–π –≥—Ä—É–ø–ø–∏—Ä–æ–≤–∫–æ–π –∫–Ω–æ–ø–æ–∫ –∏ –∫–Ω–æ–ø–∫–æ–π –≤–æ–∑–≤—Ä–∞—Ç–∞.

```php
<?php
require_once __DIR__ . '/vendor/autoload.php';
use ZenithGram\ZenithGram\ZG;
use ZenithGram\ZenithGram\Bot;
use ZenithGram\ZenithGram\Button;
use ZenithGram\ZenithGram\Enums\PaginationMode;
use ZenithGram\ZenithGram\Enums\PaginationNumberStyle;
use ZenithGram\ZenithGram\Enums\PaginationLayout;

$tg = ZG::create('–¢–û–ö–ï–ù_–ë–û–¢–ê');
$bot = new Bot($tg);

// –§—É–Ω–∫—Ü–∏—è-–ø–æ–º–æ—â–Ω–∏–∫ –¥–ª—è —Å–æ–∑–¥–∞–Ω–∏—è –∫–ª–∞–≤–∏–∞—Ç—É—Ä—ã
function getShopKeyboard(ZG $tg, int $page): array {
    // 1. –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º –¥–∞–Ω–Ω—ã–µ (30 —Ç–æ–≤–∞—Ä–æ–≤)
    $items = [];
    for ($i = 1; $i <= 30; $i++) {
        $items[] = Button::cb("üì¶ –¢–æ–≤–∞—Ä $i", "buy_$i");
    }

    // 2. –ù–∞—Å—Ç—Ä–∞–∏–≤–∞–µ–º –ø–∞–≥–∏–Ω–∞—Ü–∏—é
    return $tg->pagination()
        ->setItems($items)
        ->setPerPage(6)                         // 6 —Ç–æ–≤–∞—Ä–æ–≤ –Ω–∞ —Å—Ç—Ä–∞–Ω–∏—Ü–µ
        ->setColumns(2)                         // –í 2 –∫–æ–ª–æ–Ω–∫–∏
        ->setPage($page)                        // –¢–µ–∫—É—â–∞—è —Å—Ç—Ä–∞–Ω–∏—Ü–∞
        ->setPrefix('shop_page_')               // –ü—Ä–µ—Ñ–∏–∫—Å –Ω–∞–≤–∏–≥–∞—Ü–∏–∏
        
        // –í–∫–ª—é—á–∞–µ–º —Ä–µ–∂–∏–º –Ω–æ–º–µ—Ä–æ–≤ —Å —ç–º–æ–¥–∑–∏
        ->setMode(PaginationMode::NUMBERS)
        ->setNumberStyle(PaginationNumberStyle::EMOJI)
        ->setMaxPageBtn(5)                      // –ú–∞–∫—Å. 5 –∫–Ω–æ–ø–æ–∫ —Å—Ç—Ä–∞–Ω–∏—Ü (1Ô∏è‚É£ 2Ô∏è‚É£ 3Ô∏è‚É£ 4Ô∏è‚É£ 5Ô∏è‚É£)
        
        // –î–æ–±–∞–≤–ª—è–µ–º –∫–Ω–æ–ø–∫—É "–í –º–µ–Ω—é"
        ->addReturnBtn('üîô –í –≥–ª–∞–≤–Ω–æ–µ –º–µ–Ω—é', 'main_menu')
        
        ->create();
}

// 1. –¢–æ—á–∫–∞ –≤—Ö–æ–¥–∞: –∫–æ–º–∞–Ω–¥–∞ /shop
$bot->onBotCommand('shop', '/shop')
    ->func(function(ZG $tg) {
        $kbd = getShopKeyboard($tg, 1);
        $tg->msg("üè™ –í–∏—Ç—Ä–∏–Ω–∞ –º–∞–≥–∞–∑–∏–Ω–∞:")->inlineKbd($kbd)->send();
    });

// 2. –û–±—Ä–∞–±–æ—Ç—á–∏–∫ –Ω–∞–≤–∏–≥–∞—Ü–∏–∏ (shop_page_1, shop_page_2...)
$bot->onCallback('shop_nav', 'shop_page_%n')
    ->func(function(ZG $tg, int $page) {
        $tg->answerCallbackQuery($tg->getQueryId());
        
        $kbd = getShopKeyboard($tg, $page);
        
        // –†–µ–¥–∞–∫—Ç–∏—Ä—É–µ–º —Å–æ–æ–±—â–µ–Ω–∏–µ —Å –Ω–æ–≤–æ–π –∫–ª–∞–≤–∏–∞—Ç—É—Ä–æ–π
```
