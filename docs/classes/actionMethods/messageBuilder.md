# MessageBuilder
Trait, –∫–æ—Ç–æ—Ä—ã–π –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –≤ –∫–ª–∞—Å—Å–∞—Ö `Message` –∏ `Action` –¥–ª—è —Å–±–æ—Ä–∫–∏ –∏ –æ—Ç–ø—Ä–∞–≤–∫–∏ —Å–æ–æ–±—â–µ–Ω–∏–π

–û–Ω –≤–∫–ª—é—á–∞–µ—Ç –≤ —Å–µ–±—è –≤—Å–µ –º–µ—Ç–æ–¥—ã –¥–ª—è —Å–±–æ—Ä–∫–∏ —Å–æ–æ–±—â–µ–Ω–∏–π, –Ω–∞–ø—Ä–∏–º–µ—Ä: `text()`, `parseMode()`, `reply()`.

## –ü–æ–¥—Ä–æ–±–Ω–µ–µ –ø—Ä–æ –∫–ª–∞—Å—Å Message –∏ –¥–æ—Å—Ç—É–ø–Ω—ã—Ö –º–µ—Ç–æ–¥–∞—Ö [–∑–¥–µ—Å—å](/classes/message.md)

## –ü—Ä–∏–º–µ—Ä—ã –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è
```php
<?php
require_once __DIR__ . '/vendor/autoload.php';
use ZenithGram\ZenithGram\ZG;
use ZenithGram\ZenithGram\Bot;
use ZenithGram\ZenithGram\Button;
use ZenithGram\ZenithGram\Enums\MessageParseMode;
use ZenithGram\ZenithGram\Enums\MessageDice;

$tg = ZG::create('–¢–û–ö–ï–ù_–ë–û–¢–ê');
$bot = new Bot($tg);

// 1. –¢–µ–∫—Å—Ç —Å —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ–º
// –ò—Å–ø–æ–ª—å–∑—É–µ–º text() –∏ parseMode() –¥–ª—è –æ—Ç–ø—Ä–∞–≤–∫–∏ HTML
$bot->onBotCommand('start', '/start')
    ->text("–ü—Ä–∏–≤–µ—Ç! <b>–î–æ–±—Ä–æ –ø–æ–∂–∞–ª–æ–≤–∞—Ç—å</b> –≤ –±–æ—Ç–∞.")
    ->parseMode(MessageParseMode::HTML);

// 2. –ú–µ–¥–∏–∞ —Å –ø–æ–¥–ø–∏—Å—å—é –∏ –∫–ª–∞–≤–∏–∞—Ç—É—Ä–æ–π
// text() –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ —Å—Ç–∞–Ω–æ–≤–∏—Ç—Å—è –ø–æ–¥–ø–∏—Å—å—é (caption), –µ—Å–ª–∏ –¥–æ–±–∞–≤–ª–µ–Ω–æ –º–µ–¥–∏–∞
$bot->onText('cat', '–ø–æ–∫–∞–∂–∏ –∫–æ—Ç–∞')
    ->img('https://cataas.com/cat')
    ->text('–í–æ—Ç —Ç–≤–æ–π —Å–ª—É—á–∞–π–Ω—ã–π –∫–æ—Ç–∏–∫ üê±')
    ->inlineKbd([
        [Button::cb('–ï—â—ë –æ–¥–Ω–æ–≥–æ', 'cat')]
    ]);

// 3. –û—Ç–≤–µ—Ç –Ω–∞ —Å–æ–æ–±—â–µ–Ω–∏–µ (Reply) –∏ —É–¥–∞–ª–µ–Ω–∏–µ –∫–ª–∞–≤–∏–∞—Ç—É—Ä—ã
// reply() –∑–∞—Å—Ç–∞–≤–ª—è–µ—Ç –±–æ—Ç–∞ –æ—Ç–≤–µ—Ç–∏—Ç—å –Ω–∞ —Å–æ–æ–±—â–µ–Ω–∏–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
$bot->onText('close', '–∑–∞–∫—Ä—ã—Ç—å –º–µ–Ω—é')
    ->reply() // –û—Ç–≤–µ—Ç –Ω–∞ —Å–æ–æ–±—â–µ–Ω–∏–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
    ->text('–ö–ª–∞–≤–∏–∞—Ç—É—Ä–∞ —Å–∫—Ä—ã—Ç–∞.')
    ->removeKbd();

// 4. –û—Ç–ø—Ä–∞–≤–∫–∞ –¥–æ–∫—É–º–µ–Ω—Ç–∞ —Å –¥–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–º–∏ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º–∏
// params() –ø–æ–∑–≤–æ–ª—è–µ—Ç –ø–µ—Ä–µ–¥–∞—Ç—å —Å–ø–µ—Ü–∏—Ñ–∏—á–Ω—ã–µ —Ñ–ª–∞–≥–∏ API, –Ω–∞–ø—Ä–∏–º–µ—Ä disable_notification
$bot->onBotCommand('rules', '/rules')
    ->doc('rules.pdf')
    ->text('–û–∑–Ω–∞–∫–æ–º—å—Ç–µ—Å—å —Å –ø—Ä–∞–≤–∏–ª–∞–º–∏.')
    ->params(['disable_notification' => true]);

// 5. –†–∞–∑–≤–ª–µ–∫–∞—Ç–µ–ª—å–Ω—ã–µ –º–µ—Ç–æ–¥—ã (–ö—É–±–∏–∫)
// dice() –æ—Ç–ø—Ä–∞–≤–ª—è–µ—Ç –∞–Ω–∏–º–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —ç–º–æ–¥–∑–∏
$bot->onText('dice', '–∫–∏–Ω—å –∫—É–±–∏–∫')
    ->dice(MessageDice::Dice);
    
// 6. –°–ª–æ–∂–Ω–∞—è —Ü–µ–ø–æ—á–∫–∞ (Fluent Interface –≤–æ –≤—Å–µ–π –∫—Ä–∞—Å–µ)
$bot->onBotCommand('promo', '/promo')
    ->img('https://example.com/poster.jpg')      // –ö–∞—Ä—Ç–∏–Ω–∫–∞
    ->text("*–°–∫–∏–¥–∫–∞ 50%* –Ω–∞ –≤—Å–µ —Ç–æ–≤–∞—Ä—ã!")        // –¢–µ–∫—Å—Ç/–ü–æ–¥–ø–∏—Å—å
    ->parseMode(MessageParseMode::MarkdownV2)    // –§–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ
    ->reply()                                    // –†–µ–ø–ª–∞–π
    ->inlineKbd([[Button::url('–ö—É–ø–∏—Ç—å', 'https://example.com')]]) // –ö–Ω–æ–ø–∫–∞
    ->params(['protect_content' => true]);       // –ó–∞–ø—Ä–µ—Ç –ø–µ—Ä–µ—Å—ã–ª–∫–∏

$bot->run();
```
